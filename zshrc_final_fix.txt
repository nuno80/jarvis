# Configurazione ZSH per JARVIS
export PATH="/home/nuno/.local/bin:$PATH"

# --- JARVIS ALIASES ---
alias jarvis-start='nohup /home/nuno/.local/bin/zeroclaw -w /mnt/c/Users/Nuno/Documents/programmazione/11_JARVIS_project/jarvis_local > /home/nuno/zeroclaw.log 2>&1 &'
alias jarvis-stop='pkill zeroclaw'
alias jarvis-logs='tail -f /home/nuno/zeroclaw.log'
alias jarvis-restart='jarvis-stop 2>/dev/null; sleep 1; jarvis-start'

# --- ADB BRIDGE ---
export WIN_IP=$(ip route show default | awk "{print \$3}")
export ADB_SERVER_SOCKET=tcp:$WIN_IP:5037
